   16 {-|
   17 Função principal de atualização do estado do jogo
   18 -}
   19 atualizaJogo :: Tempo -> Jogo -> Jogo
   20 atualizaJogo tempo jogo = jogoAtualizado
   21   where
   22     jogoComTorresAtualizadas = atualizaTorres tempo jogo
   23     jogoComInimigosAtualizados = atualizaInimigos tempo jogoComTorresAtualizadas
   24     jogoComPortaisAtualizados = atualizaPortais tempo jogoComInimigosAtualizados
   25     jogoAtualizado = jogoComPortaisAtualizados
   26 
   27 {-|
   28 Atualiza o estado das Torres (incluindo ataques a Inimigos)
   29 -}
   30 atualizaTorres :: Tempo -> Jogo -> Jogo
   31 atualizaTorres tempo jogo = jogo { torresJogo = novasTorres, inimigosJogo = inimigosAtualizados }
   32   where
   33     novasTorres = map (atualizaTempoTorre tempo) (torresJogo jogo)
   34     inimigosAlvo = inimigosJogo jogo
   35     inimigosAtualizados = foldl (\inims torre -> disparaTorre torre inims) inimigosAlvo novasTorres
   36 
   37 {-|
   38 Atualiza o tempo restante de recarga de cada Torre
   39 -}
   40 atualizaTempoTorre :: Tempo -> Torre -> Torre
   41 atualizaTempoTorre tempo torre@(Torre {tempoTorre = t})
   42   | t > 0     = torre {tempoTorre = t - tempo}
   43   | otherwise = torre
   44 
   45 {-|
   46 Função que dispara a torre, aplicando as regras de rajada e ciclo de tempo
   47 -}
   48 disparaTorre :: Torre -> [Inimigo] -> [Inimigo]
   49 disparaTorre torre inimigos
   50   | tempoTorre torre <= 0 =
   51       let inimigosNoRaio = take (rajadaTorre torre) (inimigosNoAlcance torre inimigos)  
   52           -- Aplica dano a cada inimigo no alcance da torre
   53           inimigosComDano = map (atingeInimigo torre) inimigosNoRaio
   54           -- Remove os inimigos atingidos usando a função auxiliar
   55           inimigosRestantes = removeAtingidos inimigosNoRaio inimigos
   56       in inimigosComDano ++ inimigosRestantes
   57   | otherwise = inimigos
   58 
   59 {-|
   60 Função auxiliar que remove os inimigos atingidos da lista de todos os inimigos
   61 -}
   62 removeAtingidos :: [Inimigo] -> [Inimigo] -> [Inimigo]
   63 removeAtingidos atingidos todos = foldl (\acc inimigo -> delete inimigo acc) todos atingidos
   64 
   65 {-|
   66 Atualiza os créditos da base com base no butim dos inimigos derrotados
   67 -}
   68 atualizaCreditos :: [Inimigo] -> Base -> Base
   69 atualizaCreditos inimigos base =
   70     let creditosAdicionais = sum (map butimInimigo (filter (\inimigo -> vidaInimigo inimigo <= 0) inimigos))
   71     in base { creditosBase = creditosBase base + creditosAdicionais }
   72 
   73 {-|
   74 Atualiza o estado dos inimigos e os créditos da base
   75 -}
   76 atualizaInimigos :: Tempo -> Jogo -> Jogo
   77 atualizaInimigos tempo jogo =
   78     let basePos = posicaoBase (baseJogo jogo)
   79         mapa = mapaJogo jogo
   80         torres = torresJogo jogo
   81         -- Atualiza os inimigos
   82         inimigosAtualizados = map (atualizaInimigo tempo mapa torres basePos) (inimigosJogo jogo)
   83         -- Filtra apenas inimigos vivos (vida > 0)
   84         inimigosVivos = filter (\inimigo -> vidaInimigo inimigo > 0) inimigosAtualizados
   85         -- Atualiza os créditos da base com base nos inimigos derrotados
   86         baseAtualizada = atualizaCreditos inimigosAtualizados (baseJogo jogo)
   87         -- Calcula o dano da base somente para os inimigos que chegaram à base
   88         inimigosNaBase = filter (\inimigo -> vidaInimigo inimigo > 0 && chegouBase basePos inimigo) inimigosAtualizados
   89         danoTotal = sum (map (danoSeNaBase basePos) inimigosNaBase)
   90         novaVidaBase = vidaBase (baseJogo jogo) - danoTotal
   91     in jogo
   92         { inimigosJogo = inimigosVivos,  -- Atualiza com apenas inimigos vivos
   93           baseJogo = baseAtualizada { vidaBase = novaVidaBase }
   94         }

   96 {-|
   97 Atualiza o estado de um inimigo com base no tempo, mapa, torres e posição da base
   98 -}
   99 atualizaInimigo :: Tempo -> Mapa -> [Torre] -> Posicao -> Inimigo -> Inimigo
  100 atualizaInimigo tempo mapa torres basePos inimigo
  101   | chegouBase basePos inimigo = inimigo
  102   | otherwise = moveInimigo tempo mapa torres inimigo
  103 
  104 {-|
  105 Move o inimigo com base no tempo, mapa, torres e efeitos dos projéteis, ajustando posição e direção
  106 -}
  107 moveInimigo :: Tempo -> Mapa -> [Torre] -> Inimigo -> Inimigo
  108 moveInimigo tempo mapa torres inimigo@(Inimigo {velocidadeInimigo = v, posicaoInimigo = (x, y), direcaoInimigo = dir, projeteisInimigo = proj}) =
  109     let
  110         -- Verifica se está próximo da borda do bloco atual com uma margem maior
  111         (blocoX, blocoY) = (floor x, floor y)
  112         estaPertoDaBorda = case dir of
  113             Norte -> y - fromIntegral blocoY < 0.5
  114             Sul   -> fromIntegral (blocoY + 1) - y < 1.0
  115             Este  -> fromIntegral (blocoX + 1) - x < 1.0
  116             Oeste -> x - fromIntegral blocoX < 0.1
  117 
  118         -- Verifica se o próximo bloco na direção atual é Terra
  119         proximoTerreno = case dir of
  120             Norte -> acharTerreno mapa (blocoX, blocoY - 1)
  121             Sul   -> acharTerreno mapa (blocoX, blocoY + 1)
  122             Este  -> acharTerreno mapa (blocoX + 1, blocoY)
  123             Oeste -> acharTerreno mapa (blocoX - 1, blocoY)
  124 
  125         -- Decide se precisa de mudar a direção
  126         precisaMudarDirecao = estaPertoDaBorda && proximoTerreno /= Just Terra
  127 
  128         -- Calcula nova direção se necessário
  129         novaDirecao = if precisaMudarDirecao
  130                      then escolheNovaDirecao mapa (x, y) dir
  131                      else dir
  132 
  133         -- Calcula velocidade baseada nos efeitos
  134         velocidade
  135           | any (\p -> tipoProjetil p == Gelo) proj = 0
  136           | any (\p -> tipoProjetil p == Resina) proj = 0.001 
  137           | otherwise = 0.002 * v
  138 
  139         -- Calcula nova posição
  140         novaPosicao = if any (\p -> tipoProjetil p == Gelo) proj
  141                      then (x, y)
  142                      else case novaDirecao of
  143                           Norte -> (x, y - velocidade)
  144                           Sul   -> (x, y + velocidade)
  145                           Este  -> (x + velocidade, y)
  146                           Oeste -> (x - velocidade, y)
  147 
  148         -- Verifica se o inimigo está no alcance de alguma torre
  149         inimigoEstaNoAlcance = any (\torre -> not (null (inimigosNoAlcance torre [inimigo]))) torres
  150 
  151         -- Atualiza a lista de projéteis
  152         atualizaListaProjeteis :: [Projetil] -> [Projetil]
  153         atualizaListaProjeteis [] = []
  154         atualizaListaProjeteis (p:ps)
  155             | inimigoEstaNoAlcance = atualizarProjeteisInimigo ps p
  156             | otherwise = ps 
  157     in inimigo {
  158         posicaoInimigo = novaPosicao,
  159         direcaoInimigo = novaDirecao,
  160         projeteisInimigo = atualizaListaProjeteis proj
  161     }

  163 {-|
  164 Calcula a nova direção para o inimigo
  165 -}
  166 calculaProximoBloco :: Mapa -> Posicao -> Direcao -> Direcao
  167 calculaProximoBloco mapa (x, y) direcao =
  168     let novaPosicao = proximaPosicao (x, y) direcao
  169         posicaoInteira = (floor (fst novaPosicao), floor (snd novaPosicao))
  170     in case acharTerreno mapa posicaoInteira of
  171         Just Terra -> direcao
  172         _          -> escolheNovaDirecao mapa (x, y) direcao

  174 {-|
  175 Determina se um inimigo alcançou a base
  176 -}
  177 chegouBase :: Posicao -> Inimigo -> Bool
  178 chegouBase basePos inimigo = encontraBase basePos (posicaoInimigo inimigo)
  179 
  180 {-|
  181 Determina o dano que um inimigo causa à base
  182 -}
  183 danoSeNaBase :: Posicao -> Inimigo -> Float
  184 danoSeNaBase basePos inimigo
  185   | chegouBase basePos inimigo = ataqueInimigo inimigo
  186   | otherwise = 0
  187 
  188 {-|
  189 Verifica se duas posições coincidem
  190 -}
  191 encontraBase :: Posicao -> Posicao -> Bool
  192 encontraBase (x1, y1) (x2, y2) =
  193     floor x1 == floor x2 && floor y1 == floor y2
  194 
  195 {-|
  196 Escolhe uma nova direção para o inimigo com base no mapa e sua posição atual
  197 -}
  198 escolheNovaDirecao :: Mapa -> Posicao -> Direcao -> Direcao
  199 escolheNovaDirecao mapa (x, y) dirAtual =
  200     let
  201         direcoesPossiveis = direcoesValidas dirAtual
  202         blocoAtual = (floor x, floor y)
  203 
  204         {-|
  205         Verifica se é possível mover para um bloco adjacente em uma direção específica dentro de um mapa
  206         -}
  207         testarDirecao :: Direcao -> Bool
  208         testarDirecao dir =
  209             let (dx, dy) = case dir of
  210                     Norte -> (0, -1)
  211                     Sul   -> (0, 1)
  212                     Este  -> (1, 0)
  213                     Oeste -> (-1, 0)
  214                 proximoBloco = (fst blocoAtual + dx, snd blocoAtual + dy)
  215             in case acharTerreno mapa proximoBloco of
  216                 Just Terra -> True
  217                 _         -> False
  218 
  219         direcoesValidas' = filter testarDirecao direcoesPossiveis
  220     in case direcoesValidas' of
  221         (d:_) -> d
  222         []    -> dirAtual
  223 
  224 {-|
  225 Obtém o tipo de terreno de uma posição no mapa, retornando `Nothing` se estiver fora dos limites
  226 -}
  227 acharTerreno :: Mapa -> (Int, Int) -> Maybe Terreno
  228 acharTerreno mapa (x, y)
  229   | x >= 0 && y >= 0 && y < length mapa && x < length (head mapa) = Just ((mapa !! y) !! x)
  230   | otherwise = Nothing
  231 
  232 {-|
  233 Calcula a próxima posição com base na direção
  234 -}
  235 proximaPosicao :: Posicao -> Direcao -> Posicao
  236 proximaPosicao (x, y) Norte = (x, y - 1)
  237 proximaPosicao (x, y) Sul   = (x, y + 1)
  238 proximaPosicao (x, y) Este  = (x + 1, y)
  239 proximaPosicao (x, y) Oeste = (x - 1, y)
  240 
  241 {-|
  242 Retorna as direções válidas, excluindo a direção oposta
  243 -}
  244 direcoesValidas :: Direcao -> [Direcao]
  245 direcoesValidas dir = case dir of
  246     Norte -> [Este, Oeste]  -- Não pode voltar para Sul
  247     Sul   -> [Este, Oeste]  -- Não pode voltar para Norte
  248     Este  -> [Norte, Sul]   -- Não pode voltar para Oeste
  249     Oeste -> [Norte, Sul]   -- Não pode voltar para Este

  251 {-|
  252 Lança novos Inimigos dos Portais
  253 -}
  254 atualizaPortais :: Tempo -> Jogo -> Jogo
  255 atualizaPortais tempo jogo =
  256   jogo
  257     { portaisJogo = novosPortais,
  258       inimigosJogo = concat novosInimigos ++ inimigosJogo jogo }
  259   where
  260     (novosPortais, novosInimigos) = unzip $ map (lancarInimigos tempo) (portaisJogo jogo)
  261 
  262 {-|
  263 Função que lança inimigos a partir de um portal com base no tempo
  264 -}
  265 lancarInimigos :: Tempo -> Portal -> (Portal, [Inimigo])
  266 lancarInimigos tempo portal@(Portal {ondasPortal = []}) = (portal, [])
  267 lancarInimigos tempo portal =
