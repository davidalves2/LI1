   16 {-|
   17 Verifica se o estado de um jogo é válido
   18 -}
   19 validaJogo :: Jogo -> Bool
   20 validaJogo jogo =
   21   validaPortais (portaisJogo jogo) (mapaJogo jogo) (baseJogo jogo) (torresJogo jogo) &&
   22   validaInimigos jogo (portaisJogo jogo) (inimigosJogo jogo) (torresJogo jogo) &&
   23   validaTorres jogo (torresJogo jogo) &&
   24   validaBase jogo (baseJogo jogo)

   26 {-|
   27 Verifica se a Base é válida
   28 -}
   29 validaBase :: Jogo -> Base -> Bool
   30 validaBase jogo base =
   31     validaBaseSobreTerraCredito (transformarMapa (mapaJogo jogo) 0) base &&
   32     validaSobreposicaoBase (torresJogo jogo) (portaisJogo jogo) base
   33 
   34 {-|
   35 Verifica se a Base não está sobreposta a uma torre ou portal
   36 -}
   37 validaSobreposicaoBase :: [Torre] -> [Portal] -> Base -> Bool
   38 validaSobreposicaoBase torres portais (Base {posicaoBase = basePos}) =
   39     not (any (`elem` allPositions) towerPositions)
   40   where
   41     allPositions = portalPositions ++ [basePos]
   42     towerPositions = map (\(Torre {posicaoTorre = pos}) -> pos) torres
   43     portalPositions = map (\(Portal {posicaoPortal = pos}) -> pos) portais
   44 
   45 {-|
   46 Verifica se a Base está colocada sobre terra e não tem crédito negativo
   47 -}
   48 validaBaseSobreTerraCredito :: [[(Terreno, (Int, Int))]] -> Base -> Bool
   49 validaBaseSobreTerraCredito terreno (Base {posicaoBase = (x,y), creditosBase = creditos}) =
   50     case acharBloco terreno (fromInteger $ floor x,fromInteger $ floor y) of
   51         Just Terra -> creditos >=0
   52         _          -> False
   53 
   54 {-|
   55 Verifica se as Torres são válidas
   56 -}
   57 validaTorres :: Jogo -> [Torre] -> Bool
   58 validaTorres _ [] = True
   59 validaTorres jogo torres =
   60        validaSobreRelvaAlcanceRajada (transformarMapa (mapaJogo jogo) 0) torres &&
   61        validaSobreposicaoTorres torres &&
   62        validaCicloTorre torres
   63 
   64 {-|
   65 Verifica se todas as Torres possuem ciclos de disparo não negativos
   66 -}
   67 validaCicloTorre :: [Torre] -> Bool
   68 validaCicloTorre [] = True
   69 validaCicloTorre (Torre {cicloTorre = ciclo} : rest) =
   70   ciclo > 0 && validaCicloTorre rest
   71 
   72 {-|
   73 Verifica se as Torres não estão sobrepostas
   74 -}
   75 validaSobreposicaoTorres :: [Torre] -> Bool
   76 validaSobreposicaoTorres [] = True -- Base case: no towers, no overlap
   77 validaSobreposicaoTorres (Torre {posicaoTorre = (x, y)} : rest) =
   78     not (any (\(Torre {posicaoTorre = (x1, y1)}) -> (floor x1, floor y1) == (floor x, floor y)) rest) &&
   79     validaSobreposicaoTorres rest
   80 
   81 {-|
   82 Verifica se as Torres estão sobre relva e se o seu alcance é um valor positivo
   83 -}
   84 validaSobreRelvaAlcanceRajada :: [[(Terreno, (Int, Int))]] -> [Torre] -> Bool
   85 validaSobreRelvaAlcanceRajada _ [] = True
   86 validaSobreRelvaAlcanceRajada terreno (Torre {posicaoTorre = (x,y), alcanceTorre = alcance, rajadaTorre = rajada}:rest) =
   87         case acharBloco terreno (fromInteger $ floor x,fromInteger $ floor y) of
   88         Just Relva -> alcance > 0 && rajada>0 && validaSobreRelvaAlcanceRajada terreno rest
   89         _          -> False

   91 {-|
   92 Verifica se os Inimigos são válidos
   93 -}
   94 validaInimigos :: Jogo -> [Portal] -> [Inimigo] -> [Torre] -> Bool
   95 validaInimigos jogo portais inimigos torres =
   96       validaPosicaoPortal portais &&
   97       inimigoSobreTerra (transformarMapa (mapaJogo jogo) 0) inimigos &&
   98       sobreposicaoTorre (transformarMapa (mapaJogo jogo) 0) inimigos torres &&
   99       all (validaProjeteis . projeteisInimigo) inimigos
  100 
  101 {-|
  102 Verifica se a lista de projéteis ativos de um Inimigo encontra-se normalizada
  103 -}
  104 validaProjeteis :: [Projetil] -> Bool
  105 validaProjeteis projeteis =
  106   tiposUnicos projeteis && combinacaoValida projeteis
  107   where
  108     -- Verifica se não há mais de um projétil do mesmo tipo
  109     tiposUnicos :: [Projetil] -> Bool
  110     tiposUnicos ps =
  111       let tipos = map tipoProjetil ps
  112        in length tipos == length (nub tipos) 
  113 
  114     -- Verifica se não há combinações inválidas de tipos de projéteis
  115     combinacaoValida :: [Projetil] -> Bool
  116     combinacaoValida ps =
  117       let tipos = map tipoProjetil ps
  118        in not (Fogo `elem` tipos && Resina `elem` tipos) && -- Fogo e Resina
  119           not (Fogo `elem` tipos && Gelo `elem` tipos)     -- Fogo e Gelo
  120 
  121 {-|
  122 Verifica se todos os Inimigos por lançar têm a posição do respetivo portal, nível de vida positivo, e lista de projéteis ativos vazia
  123 -}
  124 validaPosicaoPortal :: [Portal] -> Bool
  125 validaPosicaoPortal [] = True
  126 validaPosicaoPortal (Portal {posicaoPortal = portalPos, ondasPortal = ondas} : rest) =
  127   all validaInimigosOnda ondas && validaPosicaoPortal rest
  128   where
  129     -- Valida todos os enimigos na onda
  130     validaInimigosOnda :: Onda -> Bool
  131     validaInimigosOnda (Onda {inimigosOnda = inimigos}) =
  132       all validaInimigo inimigos
  133 
  134     -- Valida um só inimigo
  135     validaInimigo :: Inimigo -> Bool
  136     validaInimigo (Inimigo {posicaoInimigo = (x,y), vidaInimigo = vida, projeteisInimigo = projeteis}) =
  137       (fromInteger $ floor x,fromInteger $ floor y) == portalPos && vida > 0 && null projeteis
  138 
  139 {-|
  140 Verifica se todos os Inimigos em jogo encontram-se sobre terra
  141 -}
  142 inimigoSobreTerra :: [[(Terreno, (Int, Int))]] -> [Inimigo] -> Bool
  143 inimigoSobreTerra _ [] = True -- Base case: no enemies, validation passes
  144 inimigoSobreTerra terreno (Inimigo {posicaoInimigo = (x,y)} : rest) =
  145   case acharBloco terreno (fromInteger $ floor x,fromInteger $ floor y) of
  146     Just Terra -> inimigoSobreTerra terreno rest -- Enemy is on Terra, continue checking
  147     _          -> False -- Enemy is not on Terra, validation fails
  148 
  149 {-|
  150 Verifica se os Inimigos não estão sobrepostos a Torres
  151 -}
  152 sobreposicaoTorre :: [[(Terreno, (Int, Int))]] -> [Inimigo] -> [Torre] -> Bool
  153 sobreposicaoTorre _ [] _ = True -- Base case: no enemies, validation passes
  154 sobreposicaoTorre terreno (Inimigo {posicaoInimigo = (y, x), velocidadeInimigo = velocidade} : rest) torres =
  155   velocidade >= 0 && not (any (torreSobreposicao (y, x)) torres) && sobreposicaoTorre terreno rest torres
  156   where
  157     -- Verifica se a Torre sobrepõe-se à posição inimiga fornecida
  158     torreSobreposicao :: (Float, Float) -> Torre -> Bool
  159     torreSobreposicao (y1, x1) (Torre {posicaoTorre = (y2, x2)}) =
  160       floor x1 == floor x2 && floor y1 == floor y2

  162 {-|
  163 Verifica se os Portais são válidos
  164 -}
  165 validaPortais :: [Portal] -> Mapa -> Base -> [Torre] -> Bool
  166 validaPortais [] _ _ _ = False
  167 validaPortais portais mapa base torres =
  168   all (portalSobreTerra mapa) portais &&
  169   existeCaminho base portais mapa &&
  170   naoSobrepostos portais base torres &&
  171   verificaOndasAtivas portais
  172 
  173 {-|
  174 Verifica se um Portal está posicionado sobre terra
  175 -}
  176 portalSobreTerra :: Mapa -> Portal -> Bool
  177 portalSobreTerra mapa (Portal (x, y) _) = case getTerreno mapa (floor x, floor y) of
  178   Just Terra -> True
  179   _ -> False
  180 
  181 {-|
  182 Verifica se existe um caminho de terra conectando pelo menos um Portal à Base
  183 -}
  184 existeCaminho :: Base -> [Portal] -> Mapa -> Bool
  185 existeCaminho base portais mapa =
  186   any (\portal -> conectado (floor $ fst $ posicaoBase base) (floor $ snd $ posicaoBase base)
  187                            (floor $ fst $ posicaoPortal portal) (floor $ snd $ posicaoPortal portal) mapa)
  188       portais
  189 
  190 {-|
  191 Verifica se os Portais não estão sobrepostos à Base ou às Torres
  192 -}
  193 naoSobrepostos :: [Portal] -> Base -> [Torre] -> Bool
  194 naoSobrepostos portais base torres =
  195   let posicoesProibidas = posicaoBase base : map posicaoTorre torres
  196    in all (\portal -> posicaoPortal portal `notElem` posicoesProibidas) portais
  197 
  198 {-|
  199 Verifica se há, no máximo, uma onda ativa em um portal.
  200 -}
  201 ondaAtivaPorPortal :: Portal -> Bool
  202 ondaAtivaPorPortal (Portal {ondasPortal = ondas}) =
  203     let 
  204         -- Filtra ondas ativas (com entrada ou inimigos restantes)
  205         ondasAtivas = filter (\onda -> entradaOnda onda <= 0 ) ondas
  206     in 
  207         -- Verifica se há no máximo uma onda ativa
  208         length ondasAtivas <= 1
  209 
  210 {-|
  211 Verifica se todos os portais possuem no máximo uma onda ativa.
  212 -}
  213 verificaOndasAtivas :: [Portal] -> Bool
  214 verificaOndasAtivas = all ondaAtivaPorPortal

  216 {-|
  217 Obtém o terreno do mapa em uma posição específica
  218 -}
  219 getTerreno :: Mapa -> (Int, Int) -> Maybe Terreno
  220 getTerreno mapa (x, y)
  221   | x >= 0 && y >= 0 && x < length (head mapa) && y < length mapa = Just (mapa !! y !! x)
  222   | otherwise = Nothing
  223 
  224 {-|
  225 Verifica se duas posições estão conectadas no mapa.
  226 -}
  227 conectado :: Int -> Int -> Int -> Int -> Mapa -> Bool
  228 conectado x1 y1 x2 y2 mapa = busca [(x1, y1)] []
  229   where
  230     busca [] _ = False
  231     busca (atual@(x, y):resto) visitados
  232       | atual == (x2, y2) = True
  233       | atual `elem` visitados = busca resto visitados
  234       | otherwise =
  235           let vizinhos = [(x + dx, y + dy) | (dx, dy) <- [(-1, 0), (1, 0), (0, -1), (0, 1)],
  236                             case getTerreno mapa (x + dx, y + dy) of
  237                               Just Terra -> True
  238                               _ -> False]
  239            in busca (resto ++ vizinhos) (atual : visitados)

  241 {-|
  242 Transforma um mapa de terrenos em um mapa de terrenos com coordenadas
  243 -}
  244 transformarMapa :: [[Terreno]] -> Int -> [[(Terreno, (Int, Int))]]
  245 transformarMapa [] _ = []
  246 transformarMapa (linha:resto) linhaIndex =
  247   transformarMapaL linha linhaIndex 0 : transformarMapa resto (linhaIndex + 1)
  248 
  249 {-|
  250 Transforma uma linha de terrenos em uma linha de terrenos com coordenadas
  251 -}
  252 transformarMapaL :: [Terreno] -> Int -> Int -> [(Terreno, (Int, Int))]
  253 transformarMapaL [] _ _ = []
  254 transformarMapaL (h:t) linhaIndex colunaIndex =
  255   (h, (linhaIndex, colunaIndex)) : transformarMapaL t linhaIndex (colunaIndex + 1)

  257 {-|
  258 Procura um terreno em uma matriz de blocos pelas coordenadas fornecidas
  259 -}
  260 acharBloco :: [[(Terreno, (Int, Int))]] -> (Float, Float) -> Maybe Terreno
  261 acharBloco [] _ = Nothing
  262 acharBloco (linha:t) (y,x) =
  263   case acharBlocoLinha linha (y,x) of
  264     Just terreno -> Just terreno
  265     Nothing -> acharBloco t (y,x)
  266 
  267 {-|
  268 Procura um terreno em uma linha de blocos pelas coordenadas fornecidas
  269 -}
  270 acharBlocoLinha :: [(Terreno, (Int, Int))] -> (Float, Float) -> Maybe Terreno
  271 acharBlocoLinha [] _ = Nothing
  272 acharBlocoLinha ((terreno, (x1, y1)):t) (y,x)
  273   | floor x == x1 && floor y == y1 = Just terreno
  274   | otherwise = acharBlocoLinha t (y,x)
