   15 {-|
   16 Identifica quais Inimigos estão dentro do alcance de uma Torre 
   17 -}
   18 inimigosNoAlcance :: Torre -> [Inimigo] -> [Inimigo]
   19 inimigosNoAlcance torre inimigos =
   20     [inimigo | inimigo@(Inimigo {posicaoInimigo = pos, vidaInimigo = vida}) <- inimigos,
   21                distancia (posicaoTorre torre) pos <= alcanceTorre torre && vida > 0]
   22 
   23 {-|
   24 Calcula a distância euclidiana entre duas posições
   25 -}
   26 distancia :: Posicao -> Posicao -> Float
   27 distancia (x1, y1) (x2, y2) = sqrt ((x2 - x1)^2 + (y2 - y1)^2)

   33 atingeInimigo (Torre {danoTorre = dano, projetilTorre = proj}) (Inimigo {posicaoInimigo = pos, ataqueInimigo = ataq, butimInimigo = valor, velocidadeInimigo = vel, direcaoInimigo = dir, vidaInimigo = vida, projeteisInimigo = lista}) =

   42 {-|
   43 Função para atualizar a lista de projéteis do inimigo
   44 -}
   45 atualizarProjeteisInimigo :: [Projetil] -> Projetil -> [Projetil]
   46 atualizarProjeteisInimigo [] proj = [proj]
   47 atualizarProjeteisInimigo lista proj@(Projetil {tipoProjetil = tipo}) =
   48     case tipo of
   49         Fogo   -> processaFogo lista proj
   50         Gelo   -> processaGelo lista proj
   51         Resina -> processaResina lista proj
   52   where
   53     {-|
   54     Remove Fogo e Gelo se ambos existirem
   55     -}
   56     processaFogo listaProj projetil =
   57         if Gelo `elem` map tipoProjetil listaProj
   58         then filter (\p -> tipoProjetil p /= Fogo && tipoProjetil p /= Gelo) listaProj
   59         else somaOuAdiciona listaProj projetil
   60 
   61     {-|
   62     Soma durações para Gelo ou cancela com Fogo
   63     -}
   64     processaGelo listaProj projetil =
   65         if Fogo `elem` map tipoProjetil listaProj
   66         then filter (\p -> tipoProjetil p /= Fogo && tipoProjetil p /= Gelo) listaProj
   67         else somaOuAdiciona listaProj projetil
   68 
   69     {-|
   70     Duração do Fogo dobra se Resina existir, remove Resina
   71     -}
   72     processaResina listaProj projetil =
   73         if Fogo `elem` map tipoProjetil listaProj
   74         then map (\p -> if tipoProjetil p == Fogo 
   75                         then p {duracaoProjetil = dobraDuracao (duracaoProjetil p)} 
   76                         else p) 
   77                   (filter (\p -> tipoProjetil p /= Resina) listaProj)
   78         else somaOuAdiciona listaProj projetil
   79 
   80     {-|
   81     Adiciona durações para o mesmo tipo ou adicione o projétil
   82     -}
   83     somaOuAdiciona :: [Projetil] -> Projetil -> [Projetil]
   84     somaOuAdiciona [] x = [x]
   85     somaOuAdiciona (p:ps) projetil
   86         | tipoProjetil p == tipoProjetil projetil =
   87             p {duracaoProjetil = somaDuracao (duracaoProjetil p) (duracaoProjetil projetil)} : ps
   88         | otherwise = p : somaOuAdiciona ps projetil
   89 
   90 {-|
   91 Função que soma as durações de projéteis
   92 -}
   93 somaDuracao :: Duracao -> Duracao -> Duracao
   94 somaDuracao (Finita t1) (Finita t2) = Finita (t1 + t2)
   95 somaDuracao _ Infinita = Infinita
   96 somaDuracao Infinita _ = Infinita
   97 
   98 {-|
   99 Função que duplica a duração de um projétil
  100 -}
  101 dobraDuracao :: Duracao -> Duracao
  102 dobraDuracao (Finita t) = Finita (2 * t)
  103 dobraDuracao Infinita = Infinita

  105 {-|
  106 Responsável por ativar Inimigos de um Portal, transferindo Inimigos de uma onda do Portal para a lista de Inimigos ativos
  107 -}
  108 ativaInimigo :: Tempo -> Portal -> [Inimigo] -> (Portal, [Inimigo])
  109 ativaInimigo _ portal@(Portal {ondasPortal = []}) inimigos = (portal, inimigos)
  110 ativaInimigo tempo portal@(Portal {posicaoPortal = pos, ondasPortal = onda@Onda {entradaOnda = entry, tempoOnda = time, cicloOnda = ciclo, inimigosOnda = inimigosOndaAtual} : resto}) inimigos =
  111   if null inimigosOndaAtual && entry <= 0 then
  112     -- A onda atual terminou, ativar a próxima onda, se existir
  113     let novoResto = case resto of
  114           (proxOnda : novasOndas) -> proxOnda {entradaOnda = max 0 (entradaOnda proxOnda - tempo)} : novasOndas
  115           [] -> []
  116     in (portal {ondasPortal = novoResto}, inimigos)
  117   else
  118     let
  119       -- Atualiza o tempo de entrada da onda atual
  120       novaEntradaOnda = max 0 (entradaOnda onda - tempo)
  121       ondaAtualizada = onda {entradaOnda = novaEntradaOnda}
  122 
  123       -- Lança um inimigo se o tempo da onda expira e a lista de inimigos não está vazia
  124       (portalAtualizado, novosInimigos) =
  125         if novaEntradaOnda == 0 && not (null inimigosOndaAtual) then
  126           let
  127             inimigoSemProjeteis = (head inimigosOndaAtual) {projeteisInimigo = []}
  128 
  129             ondaRestante = onda {inimigosOnda = tail inimigosOndaAtual, entradaOnda = ciclo}
  130           in (portal {ondasPortal = ondaRestante : resto}, inimigoSemProjeteis : inimigos)
  131         else
  132           (portal {ondasPortal = ondaAtualizada : resto}, inimigos)
  133     in
  134       (portalAtualizado, novosInimigos)

  136 {-|
  137 Decide se o jogo terminou
  138 -}
  139 terminouJogo :: Jogo -> Bool
  140 terminouJogo jogo = ganhouJogo jogo || perdeuJogo jogo
  141 
  142 {-|
  143 Verifica se o jogador ganhou o jogo
  144 -}
  145 ganhouJogo :: Jogo -> Bool
  146 ganhouJogo jogo = null (inimigosJogo jogo) && vidaBase (baseJogo jogo) > 0 && all (all (null . inimigosOnda) . ondasPortal) (portaisJogo jogo)
  147 
  148 {-|
  149 Verifica se o jogador perdeu o jogo
  150 -}
  151 perdeuJogo :: Jogo -> Bool
  152 perdeuJogo jogo = vidaBase (baseJogo jogo) <= 0
